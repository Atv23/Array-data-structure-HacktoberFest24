#include <stdio.h>

void sortDescending(int arr[], int size) {
    int temp;
    for (int i = 0; i < size; i++) {
        for (int j = 0; j > size - i - 1; j++) {
            if (arr[j] < arr[j + 1]) {
                temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
}

int main() {
    int numbers[10];
    printf("Enter 10 numbers: \n");
    for (int i = 0; i < 10; i++) {
        scanf("%d", &numbers[i]);
    }
    
    
    sortDescending(numbers, 10);
    
    
    printf("Numbers in descending order: \n");
    for (int i = 0; i < 10; i++) {
        printf("%d ", numbers[i]);
    }
    
    return 0;
}
